# Program to demonstrate if a number is even or odd:
number = int(input("Enter a number: "))
if number % 2 == 0:
    print("The number is even.")
else:
    print("The number is odd.")

# Program to demonstrate basic arithmetic operations:
first_number = int(input("Enter the first number: "))

print(f"Addition: {first_number} + {second_number} = {first_number + second_number}")

# Program to demonstrate if-else-elif Statement:
marks = int(input("Enter the marks: "))
if marks >= 90:
    print("Grade A")
elif marks >= 80:
    print("Grade B")
elif marks >= 75:
    print("Grade C")
else:
    print("Grade E")

# Program to find the sum of the digits of a number using WHILE loop:
number = input("Enter the number: ")
i = 0
result = 0
while i < len(number):
    result += int(number[i])
    i += 1
print(f"The sum of digits is: {result}")

# Program to calculate factorial of number using FOR loop:
number = int(input("Enter the number: "))
result = 1
for num in range(1, number + 1):
    result *= num
print(f"The factorial of {number} is {result}.")

# Program to find the average of first n natural numbers using FOR loop:
number = int(input("Enter the number: "))
result = 0
for num in range(1, number + 1):
    result += num
print(f"The average is {result / number}")

# Program to find the greatest of 3 numbers:
first_number = int(input("Enter the first number: "))

# Program to perform adjacent element multiplication in a tuple
tup = (2, 3, 4, 5)
result = [tup[i] * tup[i+1] for i in range(len(tup) - 1)]
print("Adjacent Multiplication:", result)

# Program to demonstrate multiple except block (In separate blocks)
try:
    x = int(input("Enter a number: "))
    y = int(input("Enter another number: "))
    print(x / y)
except ZeroDivisionError:
    print("Cannot divide by zero!")
except ValueError:
    print("Invalid input!")

# Program to demonstrate multiple except block (In single block)
try:
    x = int(input("Enter a number: "))
    y = int(input("Enter another number: "))
    print(x / y)
except (ZeroDivisionError, ValueError) as e:
    print("Error:", e)

# Program to find out factorial of a given number using normal Function
def fact(n):
    f = 1
    for i in range(1, n + 1):
        f *= i
    return f
n = int(input("Enter number: "))
print("Factorial:", fact(n))

# Program to demonstrate multiple Functions

a, b = 10, 5

# Program to raise an exception
def check_age(age):
    if age < 18:
        raise ValueError("Underage!")
    print("Eligible")
age = int(input("Enter age: "))
try:
    check_age(age)
except ValueError as e:
    print(e)

# Program to find out factorial of a given number using recursive Function
def rec_fact(n):
    return 1 if n == 0 else n * rec_fact(n - 1)
n = int(input("Enter number: "))
print("Factorial:", rec_fact(n))
